3fn radix_sort(mut arr: Vec<u32>) -> Vec<u32> {
4let mut max = 0;
5for &num in &arr {
6if num > max {
7max = num;
8}
9}
10
11let mut exp = 1;
12while max / exp > 0 {
13let mut output = vec![0; arr.len()];
14let mut count = [0; 10];
15
16for &num in &arr {
17let digit = (num / exp % 10) as usize;
18count[digit] += 1;
19}
20
21for i in 1..10 {
22count[i] += count[i - 1];
23}
24
25for i in (0..arr.len()).rev() {
26let digit = (arr[i] / exp % 10) as usize;
27output[count[digit] - 1] = arr[i];
28count[digit] -= 1;
29}
30
31arr = output;
32exp *= 10;
33}
34arr
35}

17void order_by(vector<int> &arr, int position) {

19vector<int> result(size);
20int count[10] = {0};
21
22// Conta a ocorrência de cada dígito
23for (size_t i = 0; i < size; i++) {

35result[count[pos] - 1] = arr[i];
36count[pos]--;
37}
38
39// Copia o resultado de volta para o vetor original
40for (size_t i = 0; i < size; i++) {
41arr[i] = result[i];
42}
